USE 76chevres;

CREATE TABLE user (
  id INT NOT NULL PRIMARY KEY AUTO_INCREMENT,
  pseudo VARCHAR(20),
  mail VARCHAR(50),
  profil_type VARCHAR(12) DEFAULT 'user

  id INT NOT NULL PRIMARY KEY AUTO_INCREMENT,
  personnage VARCHAR(80),
  citation VARCHAR (5000),
  like_count INT DEFAULT 0
  image LONGBLOB NULL,
  url_img VARCHAR(5000) NULL,
